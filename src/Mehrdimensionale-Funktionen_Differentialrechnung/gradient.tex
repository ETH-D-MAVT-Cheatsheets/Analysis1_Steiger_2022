\subsection{Gradient}
    \vspace{-0.75em}
    $$
        \grad(f(x,y,z)) = 
        \begin{pmatrix}
            f_x \\ f_y \\ f_z
        \end{pmatrix}
        =
        \begin{pmatrix}
            \frac{\partial}{\partial x} \\[0.4em] \frac{\partial}{\partial y} \\[0.4em] \frac{\partial}{\partial z}
        \end{pmatrix}
    $$
    \begin{itemize}
        \item Steht senkrecht auf Niveauflächen/ -linien.
        \item Zeigt in Richtung des grössten Anstiegs der Funktionswerte.
    \end{itemize}
