\subsection{Rechengesetze Exponential- und log-Funktionen}
    \begin{align*}
        B^a \cdot B^b = B^{a + b}\\
        \frac{B^a}{B^b} = B^{a - b}\\
        (B^a)^b = B^{a \cdot b}\\
        \log_B (a \cdot b ) = \log_B (a) + \log_B (b)\\
        \log_B \left( \frac{a}{b} \right) = \log_B (a) - \log_B (b)\\
        \log_B (a^r) = r \cdot \log_B (a)\\
        \textrm{Basisumrechnung: } \log_B (a) = \frac{\log(a)}{\log(u)}
    \end{align*}
